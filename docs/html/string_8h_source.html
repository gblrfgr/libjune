184lj_string_t lj_string_format(lj_string_t *fmt, ...) {
185 va_list args;
186 va_start(args, fmt);
187 va_list final_args;
188 va_copy(final_args, args);
189 lj_string_t result = lj_string_new(fmt->allocator);
190 const size_t INITIAL_TRY_LENGTH = 12;
191 lj_string_reserve(&result, INITIAL_TRY_LENGTH);
192 size_t necessary = vsnprintf(result.content_start, INITIAL_TRY_LENGTH,
193 lj_string_to_cstr(fmt), args);
194 if (necessary > INITIAL_TRY_LENGTH - 1) {
195 lj_string_reserve(&result, necessary + 1);
196 vsprintf(result.content_start, lj_string_to_cstr(fmt), final_args);
197 result.content_end = result.buffer_end;
198 } else {
199 result.content_end = result.content_start + necessary + 1;
200 }
201 return result;
202}
