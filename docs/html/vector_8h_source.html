52void lji_vector_grow(lj_vector_t *vec) {
53 ptrdiff_t empty_space = (vec->buffer_end - vec->buffer_start) -
54 (vec->content_end - vec->content_start);
55 if (empty_space < ((vec->buffer_end - vec->buffer_start) / 2)) {
56 size_t new_capacity = (vec->buffer_end - vec->buffer_start);
57 new_capacity += new_capacity / 2;
58 new_capacity += vec->element_size - (new_capacity % vec->element_size);
59 size_t content_length = vec->content_end - vec->content_start;
60 void *buffer = lj_allocate(vec->allocator, new_capacity);
61 memcpy(buffer, vec->content_start, content_length);
62 vec->buffer_start = vec->content_start = (char *)buffer;
63 vec->buffer_end = vec->buffer_start + new_capacity;
64 vec->content_end = vec->content_start + content_length;
65 } else {
66 memmove(vec->buffer_start, vec->content_start,
67 (vec->content_end - vec->content_start));
68 }
69}
