51void lji_vector_grow(lj_vector_t *vec) {
52 ptrdiff_t empty_space = (vec->buffer_end - vec->buffer_start) -
53 (vec->content_end - vec->content_start);
54 if (empty_space < ((vec->buffer_end - vec->buffer_start) / 2)) {
55 size_t new_capacity = (vec->buffer_end - vec->buffer_start);
56 new_capacity += new_capacity / 2;
57 new_capacity += vec->element_size - (new_capacity % vec->element_size);
58 size_t content_length = vec->content_end - vec->content_start;
59 void *buffer = lj_allocate(vec->allocator, new_capacity);
60 memcpy(buffer, vec->content_start, content_length);
61 vec->buffer_start = vec->content_start = (char *)buffer;
62 vec->buffer_end = vec->buffer_start + new_capacity;
63 vec->content_end = vec->content_start + content_length;
64 } else {
65 memmove(vec->buffer_start, vec->content_start,
66 (vec->content_end - vec->content_start));
67 }
68}

227bool lj_vector_erase(lj_vector_t *vec, size_t ind, void *out) {
228 if (ind == 0) {
229 lj_vector_pop_front(vec, out);
230 return true;
231 } else if (ind == lj_vector_size(vec) - 1) {
232 lj_vector_pop_back(vec, out);
233 return true;
234 } else if (ind > lj_vector_size(vec) - 1) {
235 return false;
236 } else {
237 if (out != NULL) {
238 lj_vector_get(vec, ind, out);
239 }
240 char *element_loc = vec->content_start + ind * vec->element_size;
241 char *first_after = vec->content_start + (ind + 1) * vec->element_size;
242 memmove(element_loc, first_after, vec->content_end - first_after);
243 vec->content_end -= vec->element_size;
244 return true;
